# Grouping rule: robots.txt parsers pick the single most-specific User-agent
# group per crawler and many do NOT merge duplicate groups. So every rule for
# a given UA must live in ONE block below (previously GPTBot / Google-Extended
# / PerplexityBot had a second "Allow: /ai/" block that most crawlers ignored,
# leaving the site fully blocked for them).

# Block AI/LLM training crawlers, but keep /ai/ open for AEO on the ones that
# also power answer engines.
User-agent: GPTBot
Allow: /ai/
Disallow: /

User-agent: Google-Extended
Allow: /ai/
Disallow: /

# PerplexityBot does both training + search; keeping /ai/ open preserves
# answer-engine visibility while blocking training scrapes.

Allow: /ai/
Disallow: /

# Default policy for every other crawler (Googlebot, Bingbot, DuckDuckBot,
# ChatGPT-User, OAI-SearchBot, etc.)
User-agent: *
Allow: /
Disallow: /api/
# Block legacy parameterised search URLs — redirected to /search/:slug
Disallow: /?q=
Disallow: /?*q=
# Block private/admin surface
Disallow: /admin/
Disallow: /profile/
Disallow: /saved/
Disallow: /reset-password/
